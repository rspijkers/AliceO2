48 template <o2::track::pid_constants::ID particle>
49 struct QATrackingEfficiencyPt {
50  static constexpr PDG_t PDGs[5] = {kElectron, kMuonMinus, kPiPlus, kKPlus, kProton};
51  static_assert(particle < 5 && "Maximum of particles reached");
52  static constexpr int particlePDG = PDGs[particle];
53  o2fw::Configurable<float> etaMin{"eta-min", -3.f, "Lower limit in eta"};
54  o2fw::Configurable<float> etaMax{"eta-max", 3.f, "Upper limit in eta"};
55  o2fw::Configurable<float> phiMin{"phi-min", 0.f, "Lower limit in phi"};
56  o2fw::Configurable<float> phiMax{"phi-max", 2.f * M_PI, "Upper limit in phi"};
57  o2fw::Configurable<float> ptMin{"pt-min", 0.f, "Lower limit in pT"};
58  o2fw::Configurable<float> ptMax{"pt-max", 5.f, "Upper limit in pT"};
59  o2fw::Configurable<int> ptBins{"pt-bins", 500, "Number of pT bins"};
60  o2fw::Configurable<int> selPrim{"sel-prim", 1, "1 select primaries, 0 select all particles"};
61  o2fw::Configurable<int> makeEff{"make-eff", 0, "Flag to produce the efficiency with TEfficiency"};
62 
63  o2fw::OutputObj<TList> list{"Efficiency"};
64  o2fw::HistogramRegistry histos{"Histos", {}, o2fw::OutputObjHandlingPolicy::AnalysisObject};
65 
66  void init(o2fw::InitContext&)
67  {
68  const TString tag = Form("%s Eta [%.2f,%.2f] Phi [%.2f,%.2f] Prim %i",
69  o2::track::pid_constants::sNames[particle],
70  etaMin.value, etaMax.value,
71  phiMin.value, phiMax.value,
72  selPrim.value);
73  const TString xaxis = "#it{p}_{T} GeV/#it{c}";
74 
75  histos.add("num", "Numerator " + tag + ";" + xaxis,
76  o2fw::kTH1D, {{ptBins.value, ptMin.value, ptMax.value}});
77 
78  histos.add("den", "Denominator " + tag + ";" + xaxis,
79  o2fw::kTH1D, {{ptBins.value, ptMin.value, ptMax.value}});
80 
81  list.setObject(new TList);
82  if (makeEff.value) {
83  list->Add(new TEfficiency("efficiency", "Efficiency " + tag + ";" + xaxis + ";Efficiency",
84  ptBins.value, ptMin.value, ptMax.value));
85  }
86  }

88  void process(const o2::soa::Join<o2::aod::Tracks, o2::aod::McTrackLabels>& tracks,
89  const o2::aod::McParticles& mcParticles)
90  {
91  std::vector<int64_t> v_reco(tracks.size());
92  int ntrks = 0;
93  for (const auto& track : tracks) {
94  const auto mcParticle = track.label();
95  if ((mcParticle.eta() < etaMin.value || mcParticle.eta() > etaMax.value)) { // Check eta
96  continue;
97  }
98  if ((mcParticle.phi() < phiMin.value || mcParticle.phi() > phiMax.value)) { // Check phi
99  continue;
100  }
101  if ((selPrim.value == 1) && (!MC::isPhysicalPrimary(mcParticles, mcParticle))) { // Requiring is physical primary
102  continue;
103  }
104  if (abs(mcParticle.pdgCode()) == particlePDG) { // Checking PDG code
105  histos.fill(HIST("num"), mcParticle.pt());
106  v_reco[ntrks++] = mcParticle.globalIndex();
107  }
108  }
109 
110  for (const auto& mcParticle : mcParticles) {
111  if ((mcParticle.eta() < etaMin.value || mcParticle.eta() > etaMax.value)) { // Check eta
112  continue;
113  }
114  if ((mcParticle.phi() < phiMin.value || mcParticle.phi() > phiMax.value)) { // Check phi
115  continue;
116  }
117  if ((selPrim.value == 1) && (!MC::isPhysicalPrimary(mcParticles, mcParticle))) { // Requiring is physical primary
118  continue;
119  }
120  if (abs(mcParticle.pdgCode()) == particlePDG) { // Checking PDG code
121  if (makeEff.value) {
122  static_cast<TEfficiency*>(list->At(0))->Fill(std::find(v_reco.begin(), v_reco.end(), mcParticle.globalIndex()) != v_reco.end(), mcParticle.pt());
123  }
124  histos.fill(HIST("den"), mcParticle.pt());
125  }
126  }
127  }
128 };
